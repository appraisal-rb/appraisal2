def for_dup
  if @options.empty?
    <<-OUTPUT.rstrip
git #{@source.inspect} do
#{indent(super)}
end
    OUTPUT
  else
    <<-OUTPUT.rstrip
git #{@source.inspect}, #{Utils.format_string(@options)} do
#{indent(super)}
end
    OUTPUT
  end
end
