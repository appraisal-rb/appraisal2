def initialize
  @appraisals = []
  @gemfile = Gemfile.new
  @gemfile.load(ENV["BUNDLE_GEMFILE"] || "Gemfile")

  raise AppraisalsNotFound unless File.exist?(path)

  run(File.read(path))
end

def appraise(name, &block)

  appraisal.instance_eval(&block)
  @appraisals << appraisal
end
